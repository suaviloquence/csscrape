csscrape -> element_list

element_list -> element element_list
             |  ""

element -> selector_list selector_ops `{` element_statement_list `}`

selector_list -> selector selector2

# TODO: pseudo-class, attr
selector -> `.` ID
         |  `#` ID
         |  ID
         |  `*`

selector2 -> selector_combinator selector selector2
          | ""

selector_combinator -> `+`
                    |  `>`
                    |  `~`
                    |  WHITESPACE
                    |  `,`  # can we wrap in parentheses?  i don't think so.
                    |  ""

selector_opts -> `?`
              |  `[]`
              |  ""

element_statement_list -> element element_statement_list
                       |  statement element_statement_list
                       |  ""

statement -> `@` ID : ID filter_list

filter_list -> `|` ID `(` arg_list `)`
            |  ""

arg_list ->	arg arg_list2
         |  ""
# allow optional trailing comma
arg_list_2 -> `,` arg_list
           |  ""

arg         ->  ID
            |   INT
            |   FLOAT
            |   STRING
